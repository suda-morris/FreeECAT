204 void ec_fsm_master_state_start(
205  ec_fsm_master_t *fsm
206  )
207 {
208  ec_master_t *master = fsm->master;
209 
210  fsm->idle = 1;
211 
212  // check for emergency requests
213  if (!list_empty(&master->emerg_reg_requests)) {
214  ec_reg_request_t *request;
215 
216  // get first request
217  request = list_entry(master->emerg_reg_requests.next,
218  ec_reg_request_t, list);
219  list_del_init(&request->list); // dequeue
220  request->state = EC_INT_REQUEST_BUSY;
221 
222  if (request->transfer_size > fsm->datagram->mem_size) {
223  EC_MASTER_ERR(master, "Emergency request data too large!\n");
224  request->state = EC_INT_REQUEST_FAILURE;
225  wake_up_all(&master->request_queue);
226  fsm->state(fsm); // continue
227  return;
228  }
229 
230  if (request->dir != EC_DIR_OUTPUT) {
231  EC_MASTER_ERR(master, "Emergency requests must be"
232  " write requests!\n");
233  request->state = EC_INT_REQUEST_FAILURE;
234  wake_up_all(&master->request_queue);
235  fsm->state(fsm); // continue
236  return;
237  }
238 
239  EC_MASTER_DBG(master, 1, "Writing emergency register request...\n");
240  ec_datagram_apwr(fsm->datagram, request->ring_position,
241  request->address, request->transfer_size);
242  memcpy(fsm->datagram->data, request->data, request->transfer_size);
243  fsm->datagram->device_index = EC_DEVICE_MAIN;
244  request->state = EC_INT_REQUEST_SUCCESS;
245  wake_up_all(&master->request_queue);
246  return;
247  }
248 
249  // check for detached config requests
250  ec_master_expire_slave_config_requests(fsm->master);
251 
252  ec_datagram_brd(fsm->datagram, 0x0130, 2);
253  ec_datagram_zero(fsm->datagram);
254  fsm->datagram->device_index = fsm->dev_idx;
255  fsm->state = ec_fsm_master_state_broadcast;
256 }

264 void ec_fsm_master_state_broadcast(
265  ec_fsm_master_t *fsm
266  )
267 {
268  ec_datagram_t *datagram = fsm->datagram;
269  unsigned int i, size;
270  ec_slave_t *slave;
271  ec_master_t *master = fsm->master;
272 
273  // bus topology change?
274  if (datagram->working_counter != fsm->slaves_responding[fsm->dev_idx]) {
275  fsm->rescan_required = 1;
276  fsm->slaves_responding[fsm->dev_idx] = datagram->working_counter;
277  EC_MASTER_INFO(master, "%u slave(s) responding on %s device.\n",
278  fsm->slaves_responding[fsm->dev_idx],
279  ec_device_names[fsm->dev_idx != 0]);
280  }
281 
282  if (fsm->link_state[fsm->dev_idx] &&
283  !master->devices[fsm->dev_idx].link_state) {
284  ec_device_index_t dev_idx;
285 
286  EC_MASTER_DBG(master, 1, "Master state machine detected "
287  "link down on %s device. Clearing slave list.\n",
288  ec_device_names[fsm->dev_idx != 0]);
289 
290 #ifdef EC_EOE
291  ec_master_eoe_stop(master);
292  ec_master_clear_eoe_handlers(master);
293 #endif
294  ec_master_clear_slaves(master);
295 
296  for (dev_idx = EC_DEVICE_MAIN;
297  dev_idx < ec_master_num_devices(master); dev_idx++) {
298  fsm->slave_states[dev_idx] = 0x00;
299  fsm->slaves_responding[dev_idx] = 0; /* Reset to trigger rescan on
300  next link up. */
301  }
302  }
303  fsm->link_state[fsm->dev_idx] = master->devices[fsm->dev_idx].link_state;
304 
305  if (datagram->state == EC_DATAGRAM_RECEIVED &&
306  fsm->slaves_responding[fsm->dev_idx]) {
307  uint8_t states = EC_READ_U8(datagram->data);
308  if (states != fsm->slave_states[fsm->dev_idx]) {
309  // slave states changed
310  char state_str[EC_STATE_STRING_SIZE];
311  fsm->slave_states[fsm->dev_idx] = states;
312  ec_state_string(states, state_str, 1);
313  EC_MASTER_INFO(master, "Slave states on %s device: %s.\n",
314  ec_device_names[fsm->dev_idx != 0], state_str);
315  }
316  } else {
317  fsm->slave_states[fsm->dev_idx] = 0x00;
318  }
319 
320  fsm->dev_idx++;
321  if (fsm->dev_idx < ec_master_num_devices(master)) {
322  // check number of responding slaves on next device
323  fsm->state = ec_fsm_master_state_start;
324  fsm->state(fsm); // execute immediately
325  return;
326  }
327 
328  if (fsm->rescan_required) {
329  down(&master->scan_sem);
330  if (!master->allow_scan) {
331  up(&master->scan_sem);
332  } else {
333  unsigned int count = 0, next_dev_slave, ring_position;
334  ec_device_index_t dev_idx;
335 
336  master->scan_busy = 1;
337  up(&master->scan_sem);
338 
339  // clear all slaves and scan the bus
340  fsm->rescan_required = 0;
341  fsm->idle = 0;
342  fsm->scan_jiffies = jiffies;
343 
344 #ifdef EC_EOE
345  ec_master_eoe_stop(master);
346  ec_master_clear_eoe_handlers(master);
347 #endif
348  ec_master_clear_slaves(master);
349 
350  for (dev_idx = EC_DEVICE_MAIN;
351  dev_idx < ec_master_num_devices(master); dev_idx++) {
352  count += fsm->slaves_responding[dev_idx];
353  }
354 
355  if (!count) {
356  // no slaves present -> finish state machine.
357  master->scan_busy = 0;
358  wake_up_interruptible(&master->scan_queue);
359  ec_fsm_master_restart(fsm);
360  return;
361  }
362 
363  size = sizeof(ec_slave_t) * count;
364  if (!(master->slaves =
365  (ec_slave_t *) kmalloc(size, GFP_KERNEL))) {
366  EC_MASTER_ERR(master, "Failed to allocate %u bytes"
367  " of slave memory!\n", size);
368  master->scan_busy = 0;
369  wake_up_interruptible(&master->scan_queue);
370  ec_fsm_master_restart(fsm);
371  return;
372  }
373 
374  // init slaves
375  dev_idx = EC_DEVICE_MAIN;
376  next_dev_slave = fsm->slaves_responding[dev_idx];
377  ring_position = 0;
378  for (i = 0; i < count; i++, ring_position++) {
379  slave = master->slaves + i;
380  while (i >= next_dev_slave) {
381  dev_idx++;
382  next_dev_slave += fsm->slaves_responding[dev_idx];
383  ring_position = 0;
384  }
385 
386  ec_slave_init(slave, master, dev_idx, ring_position, i + 1);
387 
388  // do not force reconfiguration in operation phase to avoid
389  // unnecesssary process data interruptions
390  if (master->phase != EC_OPERATION) {
391  slave->force_config = 1;
392  }
393  }
394  master->slave_count = count;
395  master->fsm_slave = master->slaves;
396 
397  /* start with first device with slaves responding; at least one
398  * has responding slaves, otherwise count would be zero. */
399  fsm->dev_idx = EC_DEVICE_MAIN;
400  while (!fsm->slaves_responding[fsm->dev_idx]) {
401  fsm->dev_idx++;
402  }
403 
404  ec_fsm_master_enter_clear_addresses(fsm);
405  return;
406  }
407  }
408 
409  if (master->slave_count) {
410 
411  // application applied configurations
412  if (master->config_changed) {
413  master->config_changed = 0;
414 
415  EC_MASTER_DBG(master, 1, "Configuration changed.\n");
416 
417  fsm->slave = master->slaves; // begin with first slave
418  ec_fsm_master_enter_write_system_times(fsm);
419 
420  } else {
421  // fetch state from first slave
422  fsm->slave = master->slaves;
423  ec_datagram_fprd(fsm->datagram, fsm->slave->station_address,
424  0x0130, 2);
425  ec_datagram_zero(datagram);
426  fsm->datagram->device_index = fsm->slave->device_index;
427  fsm->retries = EC_FSM_RETRIES;
428  fsm->state = ec_fsm_master_state_read_al_status;
429  }
430  } else {
431  ec_fsm_master_restart(fsm);
432  }
433 }

479 int ec_fsm_master_action_process_sdo(
480  ec_fsm_master_t *fsm
481  )
482 {
483  ec_master_t *master = fsm->master;
484  ec_slave_t *slave;
485  ec_sdo_request_t *req;
486 
487  // search for internal requests to be processed
488  for (slave = master->slaves;
489  slave < master->slaves + master->slave_count;
490  slave++) {
491 
492  if (!slave->config) {
493  continue;
494  }
495 
496  if (!ec_fsm_slave_is_ready(&slave->fsm)) {
497  EC_SLAVE_DBG(slave, 1, "Busy - processing external request!\n");
498  continue;
499  }
500 
501  list_for_each_entry(req, &slave->config->sdo_requests, list) {
502  if (req->state == EC_INT_REQUEST_QUEUED) {
503 
504  if (ec_sdo_request_timed_out(req)) {
505  req->state = EC_INT_REQUEST_FAILURE;
506  EC_SLAVE_DBG(slave, 1, "Internal SDO request"
507  " timed out.\n");
508  continue;
509  }
510 
511  if (slave->current_state == EC_SLAVE_STATE_INIT) {
512  req->state = EC_INT_REQUEST_FAILURE;
513  continue;
514  }
515 
516  req->state = EC_INT_REQUEST_BUSY;
517  EC_SLAVE_DBG(slave, 1, "Processing internal"
518  " SDO request...\n");
519  fsm->idle = 0;
520  fsm->sdo_request = req;
521  fsm->slave = slave;
522  fsm->state = ec_fsm_master_state_sdo_request;
523  ec_fsm_coe_transfer(&fsm->fsm_coe, slave, req);
524  ec_fsm_coe_exec(&fsm->fsm_coe, fsm->datagram);
525  return 1;
526  }
527  }
528  }
529  return 0;
530 }

538 void ec_fsm_master_action_idle(
539  ec_fsm_master_t *fsm
540  )
541 {
542  ec_master_t *master = fsm->master;
543  ec_slave_t *slave;
544 
545  // Check for pending internal SDO requests
546  if (ec_fsm_master_action_process_sdo(fsm)) {
547  return;
548  }
549 
550  // enable processing of requests
551  for (slave = master->slaves;
552  slave < master->slaves + master->slave_count;
553  slave++) {
554  ec_fsm_slave_set_ready(&slave->fsm);
555  }
556 
557  // check, if slaves have an SDO dictionary to read out.
558  for (slave = master->slaves;
559  slave < master->slaves + master->slave_count;
560  slave++) {
561  if (!(slave->sii.mailbox_protocols & EC_MBOX_COE)
562  || (slave->sii.has_general
563  && !slave->sii.coe_details.enable_sdo_info)
564  || slave->sdo_dictionary_fetched
565  || slave->current_state == EC_SLAVE_STATE_INIT
566  || slave->current_state == EC_SLAVE_STATE_BOOT
567  || slave->current_state == EC_SLAVE_STATE_UNKNOWN
568  || jiffies - slave->jiffies_preop < EC_WAIT_SDO_DICT * HZ
569  ) continue;
570 
571  EC_SLAVE_DBG(slave, 1, "Fetching SDO dictionary.\n");
572 
573  slave->sdo_dictionary_fetched = 1;
574 
575  // start fetching SDO dictionary
576  fsm->idle = 0;
577  fsm->slave = slave;
578  fsm->state = ec_fsm_master_state_sdo_dictionary;
579  ec_fsm_coe_dictionary(&fsm->fsm_coe, slave);
580  ec_fsm_coe_exec(&fsm->fsm_coe, fsm->datagram); // execute immediately
581  fsm->datagram->device_index = fsm->slave->device_index;
582  return;
583  }
584 
585  // check for pending SII write operations.
586  if (ec_fsm_master_action_process_sii(fsm)) {
587  return; // SII write request found
588  }
589 
590  ec_fsm_master_restart(fsm);

661 void ec_fsm_master_state_read_dl_status(
662  ec_fsm_master_t *fsm
663  )
664 {
665  ec_slave_t *slave = fsm->slave;
666  ec_datagram_t *datagram = fsm->datagram;
667  unsigned int i;
668 
669  if (datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
670  return;
671  }
672 
673  if (datagram->state != EC_DATAGRAM_RECEIVED) {
674  EC_SLAVE_ERR(slave, "Failed to receive AL state datagram: ");
675  ec_datagram_print_state(datagram);
676  ec_fsm_master_restart(fsm);
677  return;
678  }
679 
680  // did the slave not respond to its station address?
681  if (datagram->working_counter != 1) {
682  // try again next time
683  ec_fsm_master_action_next_slave_state(fsm);
684  return;
685  }
686 
687  ec_slave_set_dl_status(slave, EC_READ_U16(datagram->data));
688 
689  // process port state machines
690  for (i = 0; i < EC_MAX_PORTS; i++) {
691  ec_slave_port_t *port = &slave->ports[i];
692 
693  switch (port->state) {
