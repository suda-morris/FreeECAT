149 int eccdev_open(struct inode *inode, struct file *filp)
150 {
151  ec_cdev_t *cdev = container_of(inode->i_cdev, ec_cdev_t, cdev);
152  ec_cdev_priv_t *priv;
153 
154  priv = kmalloc(sizeof(ec_cdev_priv_t), GFP_KERNEL);
155  if (!priv) {
156  EC_MASTER_ERR(cdev->master,
157  "Failed to allocate memory for private data structure.\n");
158  return -ENOMEM;
159  }
160 
161  priv->cdev = cdev;
162  priv->ctx.writable = (filp->f_mode & FMODE_WRITE) != 0;
163  priv->ctx.requested = 0;
164  priv->ctx.process_data = NULL;
165  priv->ctx.process_data_size = 0;
166 
167  filp->private_data = priv;
168 
169 #if DEBUG
170  EC_MASTER_DBG(cdev->master, 0, "File opened.\n");
171 #endif
172  return 0;
173 }

179 int eccdev_release(struct inode *inode, struct file *filp)
180 {
181  ec_cdev_priv_t *priv = (ec_cdev_priv_t *) filp->private_data;
182  ec_master_t *master = priv->cdev->master;
183 
184  if (priv->ctx.requested) {
185  ecrt_release_master(master);
186  }
187 
188  if (priv->ctx.process_data) {
189  vfree(priv->ctx.process_data);
190  }
191 
192 #if DEBUG
193  EC_MASTER_DBG(master, 0, "File closed.\n");
194 #endif
195 
196  kfree(priv);
197  return 0;
198 }

258 static int eccdev_vma_fault(
259 #if LINUX_VERSION_CODE <= KERNEL_VERSION(4,10,0)
260  struct vm_area_struct *vma,
261 #endif
262  struct vm_fault *vmf
263  )
264 {
265  unsigned long offset = vmf->pgoff << PAGE_SHIFT;
266 #if LINUX_VERSION_CODE > KERNEL_VERSION(4,10,0)
267  struct vm_area_struct* vma = vmf->vma;
268 #endif
269  ec_cdev_priv_t *priv = (ec_cdev_priv_t *) vma->vm_private_data;
270  struct page *page;
271 
272  if (offset >= priv->ctx.process_data_size) {
273  return VM_FAULT_SIGBUS;
274  }
275 
276  page = vmalloc_to_page(priv->ctx.process_data + offset);
277  if (!page) {
278  return VM_FAULT_SIGBUS;
279  }
280 
281  get_page(page);
282  vmf->page = page;
283 
284  EC_MASTER_DBG(priv->cdev->master, 1, "Vma fault, virtual_address = %p,"
285 #if LINUX_VERSION_CODE > KERNEL_VERSION(4,10,0)
286  " offset = %lu, page = %p\n",(void*)vmf->address, offset, page);
287 #else
288  " offset = %lu, page = %p\n", vmf->virtual_address, offset, page);
289 #endif
290 
291  return 0;
292 }

301 struct page *eccdev_vma_nopage(
302  struct vm_area_struct *vma,
304  unsigned long address,
305  int *type
306  )
307 {
308  unsigned long offset;
309  struct page *page = NOPAGE_SIGBUS;
310  ec_cdev_priv_t *priv = (ec_cdev_priv_t *) vma->vm_private_data;
311  ec_master_t *master = priv->cdev->master;
312 
313  offset = (address - vma->vm_start) + (vma->vm_pgoff << PAGE_SHIFT);
314 
315  if (offset >= priv->ctx.process_data_size)
316  return NOPAGE_SIGBUS;
317 
318  page = vmalloc_to_page(priv->ctx.process_data + offset);
319 
320  EC_MASTER_DBG(master, 1, "Nopage fault vma, address = %#lx,"
321  " offset = %#lx, page = %p\n", address, offset, page);
322 
323  get_page(page);
324  if (type)
325  *type = VM_FAULT_MINOR;
326 
327  return page;
