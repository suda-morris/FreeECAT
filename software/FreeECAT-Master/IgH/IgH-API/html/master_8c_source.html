685 int ec_master_enter_operation_phase(
686  ec_master_t *master
687  )
688 {
689  int ret = 0;
690  ec_slave_t *slave;
691 #ifdef EC_EOE
692  ec_eoe_t *eoe;
693 #endif
694 
695  EC_MASTER_DBG(master, 1, "IDLE -> OPERATION.\n");
696 
697  down(&master->config_sem);
698  if (master->config_busy) {
699  up(&master->config_sem);
700 
701  // wait for slave configuration to complete
702  ret = wait_event_interruptible(master->config_queue,
703  !master->config_busy);
704  if (ret) {
705  EC_MASTER_INFO(master, "Finishing slave configuration"
706  " interrupted by signal.\n");
707  goto out_return;
708  }
709 
710  EC_MASTER_DBG(master, 1, "Waiting for pending slave"
711  " configuration returned.\n");
712  } else {
713  up(&master->config_sem);
714  }
715 
716  down(&master->scan_sem);
717  master->allow_scan = 0; // 'lock' the slave list
718  if (!master->scan_busy) {
719  up(&master->scan_sem);
720  } else {
721  up(&master->scan_sem);
722 
723  // wait for slave scan to complete
724  ret = wait_event_interruptible(master->scan_queue,
725  !master->scan_busy);
726  if (ret) {
727  EC_MASTER_INFO(master, "Waiting for slave scan"
728  " interrupted by signal.\n");
729  goto out_allow;
730  }
731 
732  EC_MASTER_DBG(master, 1, "Waiting for pending"
733  " slave scan returned.\n");
734  }
735 
736  // set states for all slaves
737  for (slave = master->slaves;
738  slave < master->slaves + master->slave_count;
739  slave++) {
740  ec_slave_request_state(slave, EC_SLAVE_STATE_PREOP);
741  }
742 
743 #ifdef EC_EOE
744  // ... but set EoE slaves to OP
745  list_for_each_entry(eoe, &master->eoe_handlers, list) {
746  if (ec_eoe_is_open(eoe))
747  ec_slave_request_state(eoe->slave, EC_SLAVE_STATE_OP);
748  }
749 #endif
750 
751  master->phase = EC_OPERATION;
752  master->app_send_cb = NULL;
753  master->app_receive_cb = NULL;
754  master->app_cb_data = NULL;
755  return ret;
756 
757 out_allow:
758  master->allow_scan = 1;
759 out_return:
760  return ret;
761 }

797 #endif
798

802  }
803

806  queue_size += datagram->data_size;
807  }
808  }
809 
810 #if DEBUG_INJECT
811  EC_MASTER_DBG(master, 1, "Injecting datagrams, queue_size=%zu\n",
812  queue_size);
813 #endif

817

819  // skip datagram
820  master->ext_ring_idx_rt =

823  }

832 #endif

834  datagram->cycles_sent = 0;
835 #endif

839  }

844  datagram->name, datagram->data_size,

846  }

849  cycles_t cycles_now = get_cycles();

853 #else

856 #endif
857  {
858 #if defined EC_RT_SYSLOG || DEBUG_INJECT
859  unsigned int time_us;
860 #endif

863 
864 #if defined EC_RT_SYSLOG || DEBUG_INJECT

868  / cpu_khz;
869 #else
870  time_us = (unsigned int)
871  ((jiffies - datagram->jiffies_sent) * 1000000 / HZ);
872 #endif
873  EC_MASTER_ERR(master, "Timeout %u us: Injecting"
874  " external datagram %s size=%zu,"
875  " max_queue_size=%zu\n", time_us, datagram->name,

877 #endif
878  }
879  else {

886  }

942  )

983 
987 size_t ec_master_send_datagrams(
988  ec_master_t *master,
989  ec_device_index_t device_index
990  )
991 {
992  ec_datagram_t *datagram, *next;
993  size_t datagram_size;
994  uint8_t *frame_data, *cur_data = NULL;
995  void *follows_word;
996 #ifdef EC_HAVE_CYCLES
997  cycles_t cycles_start, cycles_sent, cycles_end;
998 #endif
999  unsigned long jiffies_sent;
1000  unsigned int frame_count, more_datagrams_waiting;
1001  struct list_head sent_datagrams;
1002  size_t sent_bytes = 0;
1003 
1004 #ifdef EC_HAVE_CYCLES
1005  cycles_start = get_cycles();
1006 #endif
1007  frame_count = 0;
1008  INIT_LIST_HEAD(&sent_datagrams);
1009 
1010  EC_MASTER_DBG(master, 2, "%s(device_index = %u)\n",
1011  __func__, device_index);
1012 
1013  do {
1014  frame_data = NULL;
1015  follows_word = NULL;
1016  more_datagrams_waiting = 0;
1017 
1018  // fill current frame with datagrams
1019  list_for_each_entry(datagram, &master->datagram_queue, queue) {
1020  if (datagram->state != EC_DATAGRAM_QUEUED ||
1021  datagram->device_index != device_index) {
1022  continue;
1023  }
1024 
1025  if (!frame_data) {
1026  // fetch pointer to transmit socket buffer
1027  frame_data =
1028  ec_device_tx_data(&master->devices[device_index]);
1029  cur_data = frame_data + EC_FRAME_HEADER_SIZE;
1030  }
1031 
1032  // does the current datagram fit in the frame?
1033  datagram_size = EC_DATAGRAM_HEADER_SIZE + datagram->data_size
1034  + EC_DATAGRAM_FOOTER_SIZE;
1035  if (cur_data - frame_data + datagram_size > ETH_DATA_LEN) {
1036  more_datagrams_waiting = 1;
1037  break;
1038  }
1039 
1040  list_add_tail(&datagram->sent, &sent_datagrams);
1041  datagram->index = master->datagram_index++;
1042 
1043  EC_MASTER_DBG(master, 2, "Adding datagram 0x%02X\n",
1044  datagram->index);
1045 
1046  // set "datagram following" flag in previous datagram
1047  if (follows_word) {
1048  EC_WRITE_U16(follows_word,
1049  EC_READ_U16(follows_word) | 0x8000);
1050  }
1051 
1052  // EtherCAT datagram header
1053  EC_WRITE_U8 (cur_data, datagram->type);
1054  EC_WRITE_U8 (cur_data + 1, datagram->index);
1055  memcpy(cur_data + 2, datagram->address, EC_ADDR_LEN);
1056  EC_WRITE_U16(cur_data + 6, datagram->data_size & 0x7FF);
1057  EC_WRITE_U16(cur_data + 8, 0x0000);
1058  follows_word = cur_data + 6;
1059  cur_data += EC_DATAGRAM_HEADER_SIZE;
1060 
1061  // EtherCAT datagram data
1062  memcpy(cur_data, datagram->data, datagram->data_size);
1063  cur_data += datagram->data_size;
1064 
1065  // EtherCAT datagram footer
1066  EC_WRITE_U16(cur_data, 0x0000); // reset working counter
1067  cur_data += EC_DATAGRAM_FOOTER_SIZE;
1068  }
1069 
1070  if (list_empty(&sent_datagrams)) {
1071  EC_MASTER_DBG(master, 2, "nothing to send.\n");
1072  break;
1073  }
1074 
1075  // EtherCAT frame header
1076  EC_WRITE_U16(frame_data, ((cur_data - frame_data
1077  - EC_FRAME_HEADER_SIZE) & 0x7FF) | 0x1000);
1078 
1079  // pad frame
1080  while (cur_data - frame_data < ETH_ZLEN - ETH_HLEN)
1081  EC_WRITE_U8(cur_data++, 0x00);
1082 
1083  EC_MASTER_DBG(master, 2, "frame size: %zu\n", cur_data - frame_data);
1084 
1085  // send frame
1086  ec_device_send(&master->devices[device_index],
1087  cur_data - frame_data);
1088  /* preamble and inter-frame gap */
1089  sent_bytes += ETH_HLEN + cur_data - frame_data + ETH_FCS_LEN + 20;
1090 #ifdef EC_HAVE_CYCLES
1091  cycles_sent = get_cycles();
1092 #endif
1093  jiffies_sent = jiffies;
1094 
1095  // set datagram states and sending timestamps
1096  list_for_each_entry_safe(datagram, next, &sent_datagrams, sent) {
1097  datagram->state = EC_DATAGRAM_SENT;
1098 #ifdef EC_HAVE_CYCLES
1099  datagram->cycles_sent = cycles_sent;
1100 #endif
1101  datagram->jiffies_sent = jiffies_sent;
1102  list_del_init(&datagram->sent); // empty list of sent datagrams
1103  }
1104 
1105  frame_count++;
1106  }
1107  while (more_datagrams_waiting && frame_count < EC_TX_RING_SIZE);
1108 
1109 #ifdef EC_HAVE_CYCLES
1110  if (unlikely(master->debug_level > 1)) {
1111  cycles_end = get_cycles();
1112  EC_MASTER_DBG(master, 0, "%s()"
1113  " sent %u frames in %uus.\n", __func__, frame_count,
1114  (unsigned int) (cycles_end - cycles_start) * 1000 / cpu_khz);
1115  }
1116 #endif
1117  return sent_bytes;
1118 }

1128 void ec_master_receive_datagrams(
1129  ec_master_t *master,
1130  ec_device_t *device,
1131  const uint8_t *frame_data,
1132  size_t size
1133  )
1134 {
1135  size_t frame_size, data_size;
1136  uint8_t datagram_type, datagram_index;
1137  unsigned int cmd_follows, matched;
1138  const uint8_t *cur_data;
1139  ec_datagram_t *datagram;
1140 
1141  if (unlikely(size < EC_FRAME_HEADER_SIZE)) {
1142  if (master->debug_level || FORCE_OUTPUT_CORRUPTED) {
1143  EC_MASTER_DBG(master, 0, "Corrupted frame received"
1144  " on %s (size %zu < %u byte):\n",
1145  device->dev->name, size, EC_FRAME_HEADER_SIZE);
1146  ec_print_data(frame_data, size);
1147  }
1148  master->stats.corrupted++;
1149 #ifdef EC_RT_SYSLOG
1150  ec_master_output_stats(master);
1151 #endif
1152  return;
1153  }
1154 
1155  cur_data = frame_data;
1156 
1157  // check length of entire frame
1158  frame_size = EC_READ_U16(cur_data) & 0x07FF;
1159  cur_data += EC_FRAME_HEADER_SIZE;
1160 
1161  if (unlikely(frame_size > size)) {
1162  if (master->debug_level || FORCE_OUTPUT_CORRUPTED) {
1163  EC_MASTER_DBG(master, 0, "Corrupted frame received"
1164  " on %s (invalid frame size %zu for "
1165  "received size %zu):\n", device->dev->name,
1166  frame_size, size);
1167  ec_print_data(frame_data, size);
1168  }
1169  master->stats.corrupted++;
1170 #ifdef EC_RT_SYSLOG
1171  ec_master_output_stats(master);
1172 #endif
1173  return;
1174  }
1175 
1176  cmd_follows = 1;
1177  while (cmd_follows) {
1178  // process datagram header
1179  datagram_type = EC_READ_U8 (cur_data);
1180  datagram_index = EC_READ_U8 (cur_data + 1);
1181  data_size = EC_READ_U16(cur_data + 6) & 0x07FF;
1182  cmd_follows = EC_READ_U16(cur_data + 6) & 0x8000;
1183  cur_data += EC_DATAGRAM_HEADER_SIZE;
1184 
1185  if (unlikely(cur_data - frame_data
1186  + data_size + EC_DATAGRAM_FOOTER_SIZE > size)) {
1187  if (master->debug_level || FORCE_OUTPUT_CORRUPTED) {
1188  EC_MASTER_DBG(master, 0, "Corrupted frame received"
1189  " on %s (invalid data size %zu):\n",
1190  device->dev->name, data_size);
1191  ec_print_data(frame_data, size);
1192  }
1193  master->stats.corrupted++;
1194 #ifdef EC_RT_SYSLOG
1195  ec_master_output_stats(master);
1196 #endif
1197  return;
1198  }
1199 
1200  // search for matching datagram in the queue
1201  matched = 0;
1202  list_for_each_entry(datagram, &master->datagram_queue, queue) {
1203  if (datagram->index == datagram_index
1204  && datagram->state == EC_DATAGRAM_SENT
1205  && datagram->type == datagram_type
1206  && datagram->data_size == data_size) {
1207  matched = 1;
1208  break;
1209  }
1210  }
1211 
1212  // no matching datagram was found
1213  if (!matched) {
1214  master->stats.unmatched++;
1215 #ifdef EC_RT_SYSLOG
1216  ec_master_output_stats(master);
1217 #endif
1218 
1219  if (unlikely(master->debug_level > 0)) {
1220  EC_MASTER_DBG(master, 0, "UNMATCHED datagram:\n");
1221  ec_print_data(cur_data - EC_DATAGRAM_HEADER_SIZE,
1222  EC_DATAGRAM_HEADER_SIZE + data_size
1223  + EC_DATAGRAM_FOOTER_SIZE);
1224 #ifdef EC_DEBUG_RING
1225  ec_device_debug_ring_print(&master->devices[EC_DEVICE_MAIN]);
1226 #endif
1227  }
1228 
1229  cur_data += data_size + EC_DATAGRAM_FOOTER_SIZE;
1230  continue;
1231  }
1232 
1233  if (datagram->type != EC_DATAGRAM_APWR &&
1234  datagram->type != EC_DATAGRAM_FPWR &&
1235  datagram->type != EC_DATAGRAM_BWR &&
1236  datagram->type != EC_DATAGRAM_LWR) {
1237  // copy received data into the datagram memory,
1238  // if something has been read
1239  memcpy(datagram->data, cur_data, data_size);
1240  }
1241  cur_data += data_size;
1242 
1243  // set the datagram's working counter
1244  datagram->working_counter = EC_READ_U16(cur_data);
1245  cur_data += EC_DATAGRAM_FOOTER_SIZE;
1246 
1247  // dequeue the received datagram
1248  datagram->state = EC_DATAGRAM_RECEIVED;
1249 #ifdef EC_HAVE_CYCLES
1250  datagram->cycles_received =
1251  master->devices[EC_DEVICE_MAIN].cycles_poll;
1252 #endif
1253  datagram->jiffies_received =
1254  master->devices[EC_DEVICE_MAIN].jiffies_poll;
1255  list_del_init(&datagram->queue);
1256  }
1257 }

1321 
1324 void ec_master_update_device_stats(
1325  ec_master_t *master
1326  )
1327 {
1328  ec_device_stats_t *s = &master->device_stats;
1329  s32 tx_frame_rate, rx_frame_rate, tx_byte_rate, rx_byte_rate, loss_rate;
1330  u64 loss;
1331  unsigned int i, dev_idx;
1332 
1333  // frame statistics
1334  if (likely(jiffies - s->jiffies < HZ)) {
1335  return;
1336  }
1337 
1338  tx_frame_rate = (s->tx_count - s->last_tx_count) * 1000;
1339  rx_frame_rate = (s->rx_count - s->last_rx_count) * 1000;
1340  tx_byte_rate = s->tx_bytes - s->last_tx_bytes;
1341  rx_byte_rate = s->rx_bytes - s->last_rx_bytes;
1342  loss = s->tx_count - s->rx_count;
1343  loss_rate = (loss - s->last_loss) * 1000;
1344 
1345  /* Low-pass filter:
1346  * Y_n = y_(n - 1) + T / tau * (x - y_(n - 1)) | T = 1
1347  * -> Y_n += (x - y_(n - 1)) / tau
1348  */
1349  for (i = 0; i < EC_RATE_COUNT; i++) {
1350  s32 n = rate_intervals[i];
1351  s->tx_frame_rates[i] += (tx_frame_rate - s->tx_frame_rates[i]) / n;
1352  s->rx_frame_rates[i] += (rx_frame_rate - s->rx_frame_rates[i]) / n;
1353  s->tx_byte_rates[i] += (tx_byte_rate - s->tx_byte_rates[i]) / n;
1354  s->rx_byte_rates[i] += (rx_byte_rate - s->rx_byte_rates[i]) / n;
1355  s->loss_rates[i] += (loss_rate - s->loss_rates[i]) / n;
1356  }
1357 
1358  s->last_tx_count = s->tx_count;
1359  s->last_rx_count = s->rx_count;
1360  s->last_tx_bytes = s->tx_bytes;
1361  s->last_rx_bytes = s->rx_bytes;
1362  s->last_loss = loss;
1363 
1364  for (dev_idx = EC_DEVICE_MAIN; dev_idx < ec_master_num_devices(master);
1365  dev_idx++) {
1366  ec_device_update_stats(&master->devices[dev_idx]);
1367  }
1368 
1369  s->jiffies = jiffies;
1370 }
