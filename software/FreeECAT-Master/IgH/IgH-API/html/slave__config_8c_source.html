242  )
243 {
244  ec_slave_t *slave;
245 
246  if (sc->slave)

253  }
254 
255  if (slave->config) {
256  EC_CONFIG_DBG(sc, 1, "Failed to attach configuration. Slave %u"

259  }
260 
261  if (
262 #ifdef EC_IDENT_WILDCARDS
263  sc->vendor_id != 0xffffffff &&
264 #endif

266  ) {

268  " for configuration (0x%08X).\n",
269  slave->ring_position, slave->sii.vendor_id, sc->vendor_id);
270  return -EINVAL;
271  }
272 
273  if (
274 #ifdef EC_IDENT_WILDCARDS
275  sc->product_code != 0xffffffff &&
276 #endif
277  slave->sii.product_code != sc->product_code
278  ) {
279  EC_CONFIG_DBG(sc, 1, "Slave %u has no matching product code (0x%08X)"

283  return -EINVAL;
284  }
285 
286  // attach slave

364  EC_CONFIG_DBG(sc, 1, "Loading default mapping for PDO 0x%04X.\n",

368  for (i = 0; i < sc->slave->sii.sync_count; i++) {
369  sync = &sc->slave->sii.syncs[i];

385  return;
