219 int ec_fsm_soe_prepare_read(
220  ec_fsm_soe_t *fsm,
221  ec_datagram_t *datagram
222  )
223 {
224  uint8_t *data;
225  ec_slave_t *slave = fsm->slave;
226  ec_master_t *master = slave->master;
227  ec_soe_request_t *request = fsm->request;
228 
229  data = ec_slave_mbox_prepare_send(slave, datagram, EC_MBOX_TYPE_SOE,
230  EC_SOE_SIZE);
231  if (IS_ERR(data)) {
232  return PTR_ERR(data);
233  }
234 
235  EC_WRITE_U8(data, OPCODE_READ_REQUEST | (request->drive_no & 0x07) << 5);
236  EC_WRITE_U8(data + 1, 1 << 6); // request value
237  EC_WRITE_U16(data + 2, request->idn);
238 
239  if (master->debug_level) {
240  EC_SLAVE_DBG(slave, 0, "SSC read request:\n");
241  ec_print_data(data, EC_SOE_SIZE);
242  }
243 
244  fsm->request->jiffies_sent = jiffies;
245  fsm->state = ec_fsm_soe_read_request;
246 
247  return 0;
248 }

285 void ec_fsm_soe_read_request(
286  ec_fsm_soe_t *fsm,
287  ec_datagram_t *datagram
288  )
289 {
290  ec_slave_t *slave = fsm->slave;
291  unsigned long diff_ms;
292 
293  if (fsm->datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
294  if (ec_fsm_soe_prepare_read(fsm, datagram)) {
295  fsm->state = ec_fsm_soe_error;
296  ec_fsm_soe_print_error(fsm);
297  }
298  return;
299  }
300 
301  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
302  fsm->state = ec_fsm_soe_error;
303  EC_SLAVE_ERR(slave, "Failed to receive SoE read request: ");
304  ec_datagram_print_state(fsm->datagram);
305  ec_fsm_soe_print_error(fsm);
306  return;
307  }
308 
309  diff_ms = (jiffies - fsm->request->jiffies_sent) * 1000 / HZ;
310 
311  if (fsm->datagram->working_counter != 1) {
312  if (!fsm->datagram->working_counter) {
313  if (diff_ms < EC_SOE_RESPONSE_TIMEOUT) {
314  // no response; send request datagram again
315  if (ec_fsm_soe_prepare_read(fsm, datagram)) {
316  fsm->state = ec_fsm_soe_error;
317  ec_fsm_soe_print_error(fsm);
318  }
319  return;
320  }
321  }
322  fsm->state = ec_fsm_soe_error;
323  EC_SLAVE_ERR(slave, "Reception of SoE read request"
324  " failed after %lu ms: ", diff_ms);
325  ec_datagram_print_wc_error(fsm->datagram);
326  ec_fsm_soe_print_error(fsm);
327  return;
328  }
329 
330  fsm->jiffies_start = fsm->datagram->jiffies_sent;
331  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
332  fsm->retries = EC_FSM_RETRIES;
333  fsm->state = ec_fsm_soe_read_check;
334 }

340 void ec_fsm_soe_read_check(
341  ec_fsm_soe_t *fsm,
342  ec_datagram_t *datagram
343  )
344 {
345  ec_slave_t *slave = fsm->slave;
346 
347  if (fsm->datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
348  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
349  return;
350  }
351 
352  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
353  fsm->state = ec_fsm_soe_error;
354  EC_SLAVE_ERR(slave, "Failed to receive SoE mailbox check datagram: ");
355  ec_datagram_print_state(fsm->datagram);
356  ec_fsm_soe_print_error(fsm);
357  return;
358  }
359 
360  if (fsm->datagram->working_counter != 1) {
361  fsm->state = ec_fsm_soe_error;
362  EC_SLAVE_ERR(slave, "Reception of SoE mailbox check"
363  " datagram failed: ");
364  ec_datagram_print_wc_error(fsm->datagram);
365  ec_fsm_soe_print_error(fsm);
366  return;
367  }
368 
369  if (!ec_slave_mbox_check(fsm->datagram)) {
370  unsigned long diff_ms =
371  (fsm->datagram->jiffies_received - fsm->jiffies_start) *
372  1000 / HZ;
373  if (diff_ms >= EC_SOE_RESPONSE_TIMEOUT) {
374  fsm->state = ec_fsm_soe_error;
375  EC_SLAVE_ERR(slave, "Timeout after %lu ms while waiting for"
376  " read response.\n", diff_ms);
377  ec_fsm_soe_print_error(fsm);
378  return;
379  }
380 
381  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
382  fsm->retries = EC_FSM_RETRIES;
383  return;
384  }
385 
386  // Fetch response
387  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
388  fsm->retries = EC_FSM_RETRIES;
389  fsm->state = ec_fsm_soe_read_response;
390 }

396 void ec_fsm_soe_read_response(
397  ec_fsm_soe_t *fsm,
398  ec_datagram_t *datagram
399  )
400 {
401  ec_slave_t *slave = fsm->slave;
402  ec_master_t *master = slave->master;
403  uint8_t *data, mbox_prot, header, opcode, incomplete, error_flag,
404  value_included;
405  size_t rec_size, data_size;
406  ec_soe_request_t *req = fsm->request;
407 
408  if (fsm->datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
409  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
410  return;
411  }
412 
413  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
414  fsm->state = ec_fsm_soe_error;
415  EC_SLAVE_ERR(slave, "Failed to receive SoE read response datagram: ");
416  ec_datagram_print_state(fsm->datagram);
417  ec_fsm_soe_print_error(fsm);
418  return;
419  }
420 
421  if (fsm->datagram->working_counter != 1) {
422  fsm->state = ec_fsm_soe_error;
423  EC_SLAVE_ERR(slave, "Reception of SoE read response failed: ");
424  ec_datagram_print_wc_error(fsm->datagram);
425  ec_fsm_soe_print_error(fsm);
426  return;
427  }
428 
429  data = ec_slave_mbox_fetch(slave, fsm->datagram, &mbox_prot, &rec_size);
430  if (IS_ERR(data)) {
431  fsm->state = ec_fsm_soe_error;
432  ec_fsm_soe_print_error(fsm);
433  return;
434  }
435 
436  if (master->debug_level) {
437  EC_SLAVE_DBG(slave, 0, "SSC read response:\n");
438  ec_print_data(data, rec_size);
439  }
440 
441  if (mbox_prot != EC_MBOX_TYPE_SOE) {
442  fsm->state = ec_fsm_soe_error;
443  EC_SLAVE_ERR(slave, "Received mailbox protocol 0x%02X as response.\n",
444  mbox_prot);
445  ec_fsm_soe_print_error(fsm);
446  return;
447  }
448 
449  if (rec_size < EC_SOE_SIZE) {
450  fsm->state = ec_fsm_soe_error;
451  EC_SLAVE_ERR(slave, "Received currupted SoE read response"
452  " (%zu bytes)!\n", rec_size);
453  ec_print_data(data, rec_size);
454  ec_fsm_soe_print_error(fsm);
455  return;
456  }
457 
458  header = EC_READ_U8(data);
459  opcode = header & 0x7;
460  incomplete = (header >> 3) & 1;
461  error_flag = (header >> 4) & 1;
462 
463  if (opcode != OPCODE_READ_RESPONSE) {
464  EC_SLAVE_ERR(slave, "Received no read response (opcode %x).\n",
465  opcode);
466  ec_print_data(data, rec_size);
467  ec_fsm_soe_print_error(fsm);
468  fsm->state = ec_fsm_soe_error;
469  return;
470  }
471 
472  if (error_flag) {
473  req->error_code = EC_READ_U16(data + rec_size - 2);
474  EC_SLAVE_ERR(slave, "Received error response:\n");
475  ec_print_soe_error(slave, req->error_code);
476  ec_fsm_soe_print_error(fsm);
477  fsm->state = ec_fsm_soe_error;
478  return;
479  } else {
480  req->error_code = 0x0000;
481  }
482 
483  value_included = (EC_READ_U8(data + 1) >> 6) & 1;
484  if (!value_included) {
485  EC_SLAVE_ERR(slave, "No value included!\n");
486  ec_fsm_soe_print_error(fsm);
487  fsm->state = ec_fsm_soe_error;
488  return;
489  }
490 
491  data_size = rec_size - EC_SOE_SIZE;
492  if (ec_soe_request_append_data(req,
493  data + EC_SOE_SIZE, data_size)) {
494  fsm->state = ec_fsm_soe_error;
495  ec_fsm_soe_print_error(fsm);
496  return;
497  }
498 
499  if (incomplete) {
500  EC_SLAVE_DBG(slave, 1, "SoE data incomplete. Waiting for fragment"
501  " at offset %zu.\n", req->data_size);
502  fsm->jiffies_start = fsm->datagram->jiffies_sent;
503  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
504  fsm->retries = EC_FSM_RETRIES;
505  fsm->state = ec_fsm_soe_read_check;
506  } else {
507  if (master->debug_level) {
508  EC_SLAVE_DBG(slave, 0, "IDN data:\n");
509  ec_print_data(req->data, req->data_size);
510  }
511 
512  fsm->state = ec_fsm_soe_end; // success
513  }
514 }

522 void ec_fsm_soe_write_next_fragment(
523  ec_fsm_soe_t *fsm,
524  ec_datagram_t *datagram
525  )
526 {
527  ec_slave_t *slave = fsm->slave;
528  ec_master_t *master = slave->master;
529  ec_soe_request_t *req = fsm->request;
530  uint8_t incomplete, *data;
531  size_t max_fragment_size, remaining_size;
532  uint16_t fragments_left;
533 
534  remaining_size = req->data_size - fsm->offset;
535  max_fragment_size = slave->configured_rx_mailbox_size - EC_SOE_HEADER_SIZE;
536  incomplete = remaining_size > max_fragment_size;
537  fsm->fragment_size = incomplete ? max_fragment_size : remaining_size;
538  fragments_left = remaining_size / fsm->fragment_size - 1;
539  if (remaining_size % fsm->fragment_size) {
540  fragments_left++;
541  }
542 
543  data = ec_slave_mbox_prepare_send(slave, datagram, EC_MBOX_TYPE_SOE,
544  EC_SOE_SIZE + fsm->fragment_size);
545  if (IS_ERR(data)) {
546  fsm->state = ec_fsm_soe_error;
547  ec_fsm_soe_print_error(fsm);
548  return;
549  }
550 
551  EC_WRITE_U8(data, OPCODE_WRITE_REQUEST | incomplete << 3 |
552  (req->drive_no & 0x07) << 5);
553  EC_WRITE_U8(data + 1, 1 << 6); // only value included
554  EC_WRITE_U16(data + 2, incomplete ? fragments_left : req->idn);
555  memcpy(data + EC_SOE_SIZE, req->data + fsm->offset, fsm->fragment_size);
556 
557  fsm->offset += fsm->fragment_size;
558  if (master->debug_level) {
559  EC_SLAVE_DBG(slave, 0, "SSC write request:\n");
560  ec_print_data(data, EC_SOE_SIZE + fsm->fragment_size);
561  }
562 
563  req->jiffies_sent = jiffies;
564  fsm->state = ec_fsm_soe_write_request;
565 }

606 void ec_fsm_soe_write_request(
607  ec_fsm_soe_t *fsm,
608  ec_datagram_t *datagram
609  )
610 {
611 
612  ec_slave_t *slave = fsm->slave;
613  unsigned long diff_ms;
614 
615  if (fsm->datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
616  ec_fsm_soe_write_next_fragment(fsm, datagram);
617  return;
618  }
619 
620  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
621  fsm->state = ec_fsm_soe_error;
622  EC_SLAVE_ERR(slave, "Failed to receive SoE write request: ");
623  ec_datagram_print_state(fsm->datagram);
624  ec_fsm_soe_print_error(fsm);
625  return;
626  }
627 
628  diff_ms = (jiffies - fsm->request->jiffies_sent) * 1000 / HZ;
629 
630  if (fsm->datagram->working_counter != 1) {
631  if (!fsm->datagram->working_counter) {
632  if (diff_ms < EC_SOE_RESPONSE_TIMEOUT) {
633  // no response; send request datagram again
634  ec_fsm_soe_write_next_fragment(fsm, datagram);
635  return;
636  }
637  }
638  fsm->state = ec_fsm_soe_error;
639  EC_SLAVE_ERR(slave, "Reception of SoE write request"
640  " failed after %lu ms: ", diff_ms);
641  ec_datagram_print_wc_error(fsm->datagram);
642  ec_fsm_soe_print_error(fsm);
643  return;
644  }
645 
646  // fragment successfully sent
647  fsm->offset += fsm->fragment_size;
648 
649  if (fsm->offset < fsm->request->data_size) {
650  // next fragment
651  fsm->retries = EC_FSM_RETRIES;
652  ec_fsm_soe_write_next_fragment(fsm, datagram);
653  } else {
654  // all fragments sent; query response
655  fsm->jiffies_start = fsm->datagram->jiffies_sent;
656  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
657  fsm->retries = EC_FSM_RETRIES;
658  fsm->state = ec_fsm_soe_write_check;
659  }
660 }

672 
673  if (fsm->datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
674  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
675  return;
676  }
677 
678  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
679  fsm->state = ec_fsm_soe_error;
680  EC_SLAVE_ERR(slave, "Failed to receive SoE write request datagram: ");
681  ec_datagram_print_state(fsm->datagram);
682  ec_fsm_soe_print_error(fsm);
683  return;
684  }
685 
686  if (fsm->datagram->working_counter != 1) {
687  fsm->state = ec_fsm_soe_error;
688  EC_SLAVE_ERR(slave, "Reception of SoE write request datagram: ");
689  ec_datagram_print_wc_error(fsm->datagram);
690  ec_fsm_soe_print_error(fsm);
691  return;
692  }
693 
694  if (!ec_slave_mbox_check(fsm->datagram)) {
695  unsigned long diff_ms =
696  (datagram->jiffies_received - fsm->jiffies_start) * 1000 / HZ;
697  if (diff_ms >= EC_SOE_RESPONSE_TIMEOUT) {
698  fsm->state = ec_fsm_soe_error;
699  EC_SLAVE_ERR(slave, "Timeout after %lu ms while waiting"
700  " for write response.\n", diff_ms);
701  ec_fsm_soe_print_error(fsm);
702  return;
703  }
704 
705  ec_slave_mbox_prepare_check(slave, datagram); // can not fail.
706  fsm->retries = EC_FSM_RETRIES;
707  return;
708  }
709 
710  // Fetch response
711  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
712  fsm->retries = EC_FSM_RETRIES;
713  fsm->state = ec_fsm_soe_write_response;
714 }

716 /*****************************************************************************/
717 
720 void ec_fsm_soe_write_response(
721  ec_fsm_soe_t *fsm,
722  ec_datagram_t *datagram
723  )
724 {
725  ec_slave_t *slave = fsm->slave;
726  ec_master_t *master = slave->master;
727  ec_soe_request_t *req = fsm->request;
728  uint8_t *data, mbox_prot, opcode, error_flag;
729  uint16_t idn;
730  size_t rec_size;
731 
732  if (fsm->datagram->state == EC_DATAGRAM_TIMED_OUT && fsm->retries--) {
733  ec_slave_mbox_prepare_fetch(slave, datagram); // can not fail.
734  return; // FIXME: request again?
735  }
736 
737  if (fsm->datagram->state != EC_DATAGRAM_RECEIVED) {
738  fsm->state = ec_fsm_soe_error;
739  EC_SLAVE_ERR(slave, "Failed to receive SoE write"
740  " response datagram: ");
741  ec_datagram_print_state(fsm->datagram);
742  ec_fsm_soe_print_error(fsm);
743  return;
744  }
745 
746  if (fsm->datagram->working_counter != 1) {
747  fsm->state = ec_fsm_soe_error;
748  EC_SLAVE_ERR(slave, "Reception of SoE write response failed: ");
749  ec_datagram_print_wc_error(fsm->datagram);
750  ec_fsm_soe_print_error(fsm);
751  return;
752  }
753 
754  data = ec_slave_mbox_fetch(slave, fsm->datagram, &mbox_prot, &rec_size);
755  if (IS_ERR(data)) {
756  fsm->state = ec_fsm_soe_error;
757  ec_fsm_soe_print_error(fsm);
758  return;
759  }
760 
761  if (master->debug_level) {
762  EC_SLAVE_DBG(slave, 0, "SSC write response:\n");
763  ec_print_data(data, rec_size);
764  }
765 
766  if (mbox_prot != EC_MBOX_TYPE_SOE) {
767  fsm->state = ec_fsm_soe_error;
768  EC_SLAVE_ERR(slave, "Received mailbox protocol 0x%02X as response.\n",
769  mbox_prot);
770  ec_fsm_soe_print_error(fsm);
771  return;
772  }
773 
774  if (rec_size < EC_SOE_SIZE) {
775  fsm->state = ec_fsm_soe_error;
776  EC_SLAVE_ERR(slave, "Received corrupted SoE write response"
777  " (%zu bytes)!\n", rec_size);
778  ec_print_data(data, rec_size);
779  ec_fsm_soe_print_error(fsm);
780  return;
781  }
782 
783  opcode = EC_READ_U8(data) & 0x7;
784  if (opcode != OPCODE_WRITE_RESPONSE) {
785  EC_SLAVE_ERR(slave, "Received no write response"
786  " (opcode %x).\n", opcode);
787  ec_print_data(data, rec_size);
788  ec_fsm_soe_print_error(fsm);
789  fsm->state = ec_fsm_soe_error;
790  return;
791  }
792 
793  idn = EC_READ_U16(data + 2);
794  if (idn != req->idn) {
795  EC_SLAVE_ERR(slave, "Received response for"
796  " wrong IDN 0x%04x.\n", idn);
797  ec_print_data(data, rec_size);
798  ec_fsm_soe_print_error(fsm);
799  fsm->state = ec_fsm_soe_error;
800  return;
801  }
802 
803  error_flag = (EC_READ_U8(data) >> 4) & 1;
804  if (error_flag) {
805  if (rec_size < EC_SOE_SIZE + 2) {
806  EC_SLAVE_ERR(slave, "Received corrupted error response"
807  " - error flag set, but received size is %zu.\n",
808  rec_size);
809  } else {
810  req->error_code = EC_READ_U16(data + EC_SOE_SIZE);
811  EC_SLAVE_ERR(slave, "Received error response:\n");
812  ec_print_soe_error(slave, req->error_code);
813  }
814  ec_print_data(data, rec_size);
815  ec_fsm_soe_print_error(fsm);
816  fsm->state = ec_fsm_soe_error;
817  } else {
818  req->error_code = 0x0000;
819  fsm->state = ec_fsm_soe_end; // success
820  }
821 }
